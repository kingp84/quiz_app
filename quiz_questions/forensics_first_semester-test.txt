# -----------------------------
# Storage for student responses
# -----------------------------
student_responses = {"mc": [], "short": [], "bonus": None, "student_info": {}}

# -----------------------------
# Utility: clear the screen
# -----------------------------
def clear_screen():
    os.system('cls' if os.name == 'nt' else 'clear')

# -----------------------------
# Welcome screen
# -----------------------------
def welcome_screen():
    print("=======================================")
    print(" Welcome to the '25-'26 1st Semester Test!")
    print("=======================================")
    name = input("Please enter your first and last name: ").strip()
    hour = input("Please enter your class hour (e.g., 1st hour, 7th hour): ").strip()
    student_responses["student_info"] = {"name": name, "hour": hour}

    print("\n--- Test Instructions ---")
    print("1. Read each multiple-choice question carefully.")
    print("2. Select the best answer from the options provided.")
    print("3. Answer short-answer questions in complete sentences.")
    print("4. The bonus question can earn up to 10 points but will not subtract points.")
    print("5. Do your best and good luck!\n")

# -----------------------------
# Pause before test
# -----------------------------
def pause_before_test():
    input("\nPress Enter to begin the test...")
    clear_screen()  # ✅ Clear screen after pressing Enter

# -----------------------------
# Multiple-choice question with validation + timestamps
# -----------------------------
def ask_mc(q):
    print("\nMultiple Choice Question:")
    print(q["multiple_choice_question"])

    options = q["options"].copy()
    random.shuffle(options)

    print("\nChoose the correct answer:")
    for i, option in enumerate(options, start=1):
        print(f"{i}. {option}")

    start_time = datetime.now()

    while True:
        choice = input("Choose the correct option (1-4): ").strip()
        if choice.isdigit() and 1 <= int(choice) <= len(options):
            choice = int(choice)
            break
        else:
            print("Invalid input. Please enter a number between 1 and 4.")

    end_time = datetime.now()

    selected = options[choice - 1]
    student_responses["mc"].append({
        "question": q["multiple_choice_question"],
        "response": selected,
        "start_time": start_time.strftime("%Y-%m-%d %H:%M:%S"),
        "end_time": end_time.strftime("%Y-%m-%d %H:%M:%S")
    })

    if selected.lower() == q["answer"].lower():
        print("✅ Correct! +4 points")
        return 4
    else:
        print("❌ Incorrect. 0 points")
        print(f"Correct answer: {q['answer']}")
        return 0

# -----------------------------
# Short-answer questions with validation + timestamps
# -----------------------------
def ask_short(short_questions):
    print("\nShort-Answer Section:")
    for q in short_questions:
        print(q["short_answer_question"])
        start_time = datetime.now()
        while True:
            response = input("Your answer: ").strip()
            if response:
                break
            else:
                print("Response cannot be empty. Please type your answer.")
        end_time = datetime.now()
        student_responses["short"].append({
            "question": q["short_answer_question"],
            "response": response,
            "start_time": start_time.strftime("%Y-%m-%d %H:%M:%S"),
            "end_time": end_time.strftime("%Y-%m-%d %H:%M:%S")
        })

# -----------------------------
# Bonus question with validation + timestamps
# -----------------------------
def ask_bonus(bonus_question):
    print("\nBonus Short-Answer Question:")
    print(bonus_question["short_answer_question"])
    start_time = datetime.now()
    while True:
        response = input("Your answer: ").strip()
        if response:
            break
        else:
            print("Response cannot be empty. Please type your answer.")
    end_time = datetime.now()
    student_responses["bonus"] = {
        "response": response,
        "start_time": start_time.strftime("%Y-%m-%d %H:%M:%S"),
        "end_time": end_time.strftime("%Y-%m-%d %H:%M:%S")
    }

    while True:
        bonus_score = input("Enter bonus points awarded (0-10): ").strip()
        if bonus_score.isdigit():
            bonus_score = int(bonus_score)
            if 0 <= bonus_score <= 10:
                break
        print("Invalid input. Please enter a number between 0 and 10.")

    print(f"Bonus points awarded: {bonus_score}")
    return bonus_score

# -----------------------------
# Show correct answers
# -----------------------------
def show_answers(mc_questions, short_questions, bonus_question):
    print("\n--- Correct Answers ---")
    for idx, q in enumerate(mc_questions, start=1):
        print(f"MC Q{idx}: {q['multiple_choice_question']}")
        print(f"Correct Answer: {q['answer']}\n")

    for idx, q in enumerate(short_questions, start=1):
        print(f"Short Q{idx}: {q['short_answer_question']}")
        print(f"Correct Answer: {q['answer']}\n")

    print(f"Bonus Q: {bonus_question['short_answer_question']}")
    print(f"Suggested Answer: {bonus_question['answer']}\n")

# -----------------------------
# Save results to CSV (per hour)
# -----------------------------
def save_responses_to_csv(score, bonus_question):
    info = student_responses["student_info"]
    name = info["name"]
    hour = info["hour"].lower()

    if "1" in hour:
        folder = "student_responses/first_hour"
        filename = "grades.csv"
    elif "7" in hour:
        folder = "student_responses/seventh_hour"
        filename = "grades.csv"
    else:
        folder = "student_responses"
        filename = "grades.csv"

    os.makedirs(folder, exist_ok=True)
    filepath = os.path.join(folder, filename)

    file_exists = os.path.isfile(filepath)
    with open(filepath, "a", newline="", encoding="utf-8") as f:
        writer = csv.writer(f)
        if not file_exists:
            writer.writerow(["Name", "Hour", "Score", "MC Responses", "Short Responses", "Bonus Response"])
        mc_resps = "; ".join([f"{mc['question']} -> {mc['response']} (Start: {mc['start_time']}, End: {mc['end_time']})" for mc in student_responses["mc"]])
        short_resps = "; ".join([f"{sa['question']} -> {sa['response']} (Start: {sa['start_time']}, End: {sa['end_time']})" for sa in student_responses["short"]])
        bonus_resp = f"{bonus_question['short_answer_question']} -> {student_responses['bonus']['response']} (Start: {student_responses['bonus']['start_time']}, End: {student_responses['bonus']['end_time']})"
        writer.writerow([info["name"], info["hour"], score, mc_resps, short_resps, bonus_resp])

    print(f"\n✅ Responses saved to {filepath}")

# -----------------------------
# Run the quiz
# -----------------------------
def run_quiz(mc_questions, short_questions, bonus_question):
    welcome_screen()
    pause_before_test()

    score = 0
    shuffled_mc = mc_questions.copy()
    random.shuffle(shuffled_mc)

    for q in shuffled_mc:
        score += ask_mc(q)

    ask_short(short_questions)
    score += ask_bonus(bonus_question)

    print("\n=======================================")
    print(f"Final Score: {score} points")
    print("=======================================")
    print("\nStudent Info:", student_responses["student_info"])
    print("\nRecorded Student Responses:")
    print(student_responses)

    show_answers(mc_questions, short_questions, bonus_question)
    save_responses_to_csv(score, bonus_question)

run_quiz(mc_questions, short_questions, bonus_question)
